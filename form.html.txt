    <script>
        // ⚠️ IMPORTANT: Yahan bhi apna Google Apps Script URL daalna!
        const API_URL = 'https://script.google.com/macros/s/AKfycbxaht4dW0J3dm8MMi0j3rFPAt-_4Xq_uWgKXg8T6ucC5ZpVBo4SWlcSrVtZ5u_pUKc/exec';

        // Auto-fill phone from URL if coming from check page
        window.onload = function() {
            const urlParams = new URLSearchParams(window.location.search);
            const phone = urlParams.get('phone');
            if (phone) {
                document.getElementById('referenceNumber').value = phone;
            }
        };

        async function submitForm(event) {
            event.preventDefault();

            const submitBtn = document.getElementById('submitBtn');
            const loadingDiv = document.getElementById('loading');
            const successDiv = document.getElementById('success');
            const errorDiv = document.getElementById('error');

            // Reset
            successDiv.classList.remove('show');
            errorDiv.classList.remove('show');

            // Collect data
            const data = {
                action: 'addSale',
                nameOnBill: document.getElementById('nameOnBill').value.trim(),
                personOnCall: document.getElementById('personOnCall').value.trim(),
                product: document.getElementById('product').value.trim(),
                agentId: document.getElementById('agentId').value.trim(),
                agentName: document.getElementById('agentName').value.trim(),
                referenceNumber: document.getElementById('referenceNumber').value.trim(),
                conf: document.getElementById('conf').value.trim(),
                status: document.getElementById('status').value
            };

            // Show loading
            loadingDiv.classList.add('show');
            submitBtn.disabled = true;

            try {
                const response = await fetch(API_URL, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify(data)
                });

                const result = await response.json();

                loadingDiv.classList.remove('show');
                submitBtn.disabled = false;

                if (result.success) {
                    successDiv.classList.add('show');
                    document.getElementById('salesForm').reset();
                    
                    // Auto-fill phone again if it was from URL
                    const urlParams = new URLSearchParams(window.location.search);
                    const phone = urlParams.get('phone');
                    if (phone) {
                        document.getElementById('referenceNumber').value = phone;
                    }
                } else {
                    errorDiv.textContent = `❌ Error: ${result.error || 'Something went wrong'}`;
                    errorDiv.classList.add('show');
                }

            } catch (err) {
                loadingDiv.classList.remove('show');
                submitBtn.disabled = false;
                errorDiv.textContent = `❌ Network Error: ${err.message}`;
                errorDiv.classList.add('show');
            }
        }
    </script>
